  <script>

    // Helper function to detect mobile devices

    function isMobile() {

      return /Mobi|Android/i.test(navigator.userAgent);

    }


    // Variables to track mobile-specific state for jiggle

    let mobileFirstTapDone = false;


    // Trigger a jiggle (rotation) on the vowel keys for 1 second

    function triggerJiggle() {

      const vowelKeysContainer = document.getElementById('vowel-keys');

      vowelKeysContainer.classList.add('jiggle');

      setTimeout(() => {

        vowelKeysContainer.classList.remove('jiggle');

      }, 1000);

    }


    // --- Mobile drag-and-drop simulation for vowel keys ---

    let dragLetter = null;

    let dragGhost = null;

    function updateDragGhost(x, y) {

      if (dragGhost) {

        dragGhost.style.left = (x - dragGhost.offsetWidth / 2) + 'px';

        dragGhost.style.top = (y - dragGhost.offsetHeight / 2) + 'px';

      }

    }

    function onVowelTouchStart(e) {

      e.preventDefault();

      dragLetter = this.dataset.letter;

      // Create a ghost copy of the key

      dragGhost = this.cloneNode(true);

      dragGhost.style.position = 'absolute';

      dragGhost.style.pointerEvents = 'none';

      dragGhost.style.opacity = '0.8';

      dragGhost.style.zIndex = '1000';

      document.body.appendChild(dragGhost);

      updateDragGhost(e.touches[0].clientX, e.touches[0].clientY);

    }

    function onVowelTouchMove(e) {

      e.preventDefault();

      updateDragGhost(e.touches[0].clientX, e.touches[0].clientY);

    }

    function onVowelTouchEnd(e) {

      if (dragGhost) {

        let touch = e.changedTouches[0];

        let target = document.elementFromPoint(touch.clientX, touch.clientY);

        if (target && target.tagName === 'INPUT' && target.classList.contains('editable')) {

          target.value = dragLetter;

          checkCompletion();

        }

        dragLetter = null;

        document.body.removeChild(dragGhost);

        dragGhost = null;

      }

    }


    // Launch a simple confetti burst on win

    function launchConfetti() {

      const confettiContainer = document.createElement('div');

      confettiContainer.id = 'confetti-container';

      confettiContainer.style.position = 'fixed';

      confettiContainer.style.top = '0';

      confettiContainer.style.left = '0';

      confettiContainer.style.width = '100%';

      confettiContainer.style.height = '100%';

      confettiContainer.style.pointerEvents = 'none';

      document.body.appendChild(confettiContainer);


      const colors = ['#FFC700', '#FF0000', '#2E3192', '#41BBC7', '#7CFC00', '#FF69B4'];

      const numConfetti = 50;

      for (let i = 0; i < numConfetti; i++) {

        const confetti = document.createElement('div');

        confetti.classList.add('confetti');

        confetti.style.position = 'absolute';

        confetti.style.width = '8px';

        confetti.style.height = '8px';

        confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];

        confetti.style.top = Math.random() * 100 + '%';

        confetti.style.left = Math.random() * 100 + '%';

        confetti.style.opacity = Math.random();

        confetti.style.transform = `rotate(${Math.random() * 360}deg)`;

        confetti.style.transition = 'transform 2s ease-out, top 2s ease-out, opacity 2s ease-out';

        confettiContainer.appendChild(confetti);

        

        // animate confetti falling

        setTimeout(() => {

          confetti.style.top = '110%';

          confetti.style.transform = `rotate(${Math.random() * 360}deg)`;

          confetti.style.opacity = '0';

        }, 50);

      }

      

      setTimeout(() => {

        document.body.removeChild(confettiContainer);

      }, 2000);

    }


    document.addEventListener('DOMContentLoaded', () => {

      // Three preset 6×6 grids. In each grid vowels (A, E, I, O, U) are the letters the user must re–enter.

      const grids = [

        {

          rows: [

            "ATTEST",

            "SHERPA",

            "TIARAS",

            "RESORT",

            "AVERSE",

            "LESSER"

          ]

        },

        {

          rows: [

            "ACHING",

            "BOOGER",

            "SPOUSE",

            "UPDATE",

            "REINED",

            "DREARY"

          ]

        },

        {

          rows: [

            "SKIING",

            "TENNER",

            "RESCUE",

            "OPIATE",

            "MESSED",

            "ARTERY"

          ]

        }

      ];


      let currentGrid;

      let startTime;

      let timerInterval;

      let currentSelectedInput = null;


      // Initialize a new game (new grid and timer)

      function initGame() {

        clearInterval(timerInterval);

        mobileFirstTapDone = false; // Reset mobile first-tap state for jiggle

        const gridElement = document.getElementById('grid');

        gridElement.innerHTML = '';

        // Pick a random grid from the array

        currentGrid = grids[Math.floor(Math.random() * grids.length)];

        // Create the 6×6 grid

        for (let r = 0; r < 6; r++) {

          const rowStr = currentGrid.rows[r];

          for (let c = 0; c < 6; c++) {

            const letter = rowStr[c];

            const tile = document.createElement('div');

            tile.classList.add('tile');

            // For vowels, replace the letter with an input field

            if ("AEIOU".includes(letter)) {

              const input = document.createElement('input');

              input.setAttribute('maxlength', '1');

              input.dataset.solution = letter;

              input.dataset.row = r;

              input.dataset.col = c;

              input.classList.add('editable');

              input.value = '';

              // On mobile, disable keyboard input by marking inputs readonly.

              if (isMobile()) {

                input.setAttribute('readonly', 'true');

              }

              input.addEventListener('focus', () => {

                currentSelectedInput = input;

                if (isMobile()) {

                  // Only trigger jiggle if this is not the first tap

                  if (mobileFirstTapDone) {

                    triggerJiggle();

                  } else {

                    mobileFirstTapDone = true;

                  }

                }

              });

              input.addEventListener('keydown', (e) => {

                // On mobile, ignore keyboard input.

                if (isMobile()) {

                  e.preventDefault();

                  return;

                }

                const row = parseInt(input.dataset.row);

                const col = parseInt(input.dataset.col);

                if (e.key === 'ArrowRight') {

                  e.preventDefault();

                  moveSelection(row, col, 0, 1);

                } else if (e.key === 'ArrowLeft') {

                  e.preventDefault();

                  moveSelection(row, col, 0, -1);

                } else if (e.key === 'ArrowDown') {

                  e.preventDefault();

                  moveSelection(row, col, 1, 0);

                } else if (e.key === 'ArrowUp') {

                  e.preventDefault();

                  moveSelection(row, col, -1, 0);

                } else {

                  const key = e.key.toUpperCase();

                  if ("AEIOU".includes(key)) {

                    input.value = key;

                    e.preventDefault();

                    moveToNextInput();

                    checkCompletion();

                  } else if (e.key.length === 1) {

                    e.preventDefault();

                  }

                }

              });

              input.addEventListener('dragover', (e) => {

                e.preventDefault();

              });

              input.addEventListener('drop', (e) => {

                e.preventDefault();

                const droppedLetter = e.dataTransfer.getData('text/plain');

                if ("AEIOU".includes(droppedLetter)) {

                  input.value = droppedLetter;

                  checkCompletion();

                }

              });

              tile.appendChild(input);

            } else {

              // For consonants simply display the letter

              tile.textContent = letter;

            }

            gridElement.appendChild(tile);

          }

        }

        // Start/restart the timer

        startTime = Date.now();

        document.getElementById('timer').textContent = "Time: 0s";

        timerInterval = setInterval(updateTimer, 1000);

      }


      // Reset the current grid (clear inputs and restart timer) without changing the grid

      function resetCurrentGrid() {

        clearInterval(timerInterval);

        const inputs = document.querySelectorAll('input.editable');

        inputs.forEach(input => {

          input.value = '';

        });

        startTime = Date.now();

        document.getElementById('timer').textContent = "Time: 0s";

        timerInterval = setInterval(updateTimer, 1000);

      }


      // Moves focus from the current input to a neighboring input in the given direction

      function moveSelection(row, col, dRow, dCol) {

        const newRow = row + dRow;

        const newCol = col + dCol;

        const nextInput = document.querySelector(`input.editable[data-row="${newRow}"][data-col="${newCol}"]`);

        if (nextInput) {

          nextInput.focus();

        }

      }


      // Moves focus to the next empty input (in row–major order)

      function moveToNextInput() {

        const inputs = Array.from(document.querySelectorAll('input.editable'));

        inputs.sort((a, b) => {

          const aRow = parseInt(a.dataset.row),

                bRow = parseInt(b.dataset.row);

          return aRow === bRow ? parseInt(a.dataset.col) - parseInt(b.dataset.col) : aRow - bRow;

        });

        const currentIndex = inputs.indexOf(document.activeElement);

        if (currentIndex >= 0 && currentIndex < inputs.length - 1) {

          inputs[currentIndex + 1].focus();

        }

      }


      // Update the timer display each second

      function updateTimer() {

        const elapsed = Math.floor((Date.now() - startTime) / 1000);

        document.getElementById('timer').textContent = "Time: " + elapsed + "s";

      }


      // Check whether all vowel inputs are filled correctly

      function checkCompletion() {

        let allFilled = true;

        let allCorrect = true;

        const inputs = document.querySelectorAll('input.editable');

        inputs.forEach(input => {

          const val = input.value.toUpperCase();

          if (val === '' || !("AEIOU".includes(val))) {

            allFilled = false;

          }

          if (val !== input.dataset.solution) {

            allCorrect = false;

          }

        });

        if (allFilled && allCorrect) {

          clearInterval(timerInterval);

          const elapsed = Math.floor((Date.now() - startTime) / 1000);

          let speedRank;

          if (elapsed < 20) {

            speedRank = "💨Fast";

          } else if (elapsed < 60) {

            speedRank = "👌OK";

          } else {

            speedRank = "🐢Slow";

          }

          // Launch confetti burst

          launchConfetti();

          // Update modal message with line breaks

          document.getElementById('completion-message').innerHTML = 

            "Complete!<br> You took " + elapsed + "s<br> Speed rank: " + speedRank;

          document.getElementById('completion-modal').classList.remove('hidden');

        }

      }


      // Info modal handling

      const infoModal = document.getElementById('info-modal');

      const infoIcon = document.getElementById('info-icon');

      const closeInfo = document.getElementById('close-info');

      infoIcon.addEventListener('click', () => {

        infoModal.classList.remove('hidden');

      });

      closeInfo.addEventListener('click', () => {

        infoModal.classList.add('hidden');

      });


      // Completion modal handling

      const completionModal = document.getElementById('completion-modal');

      const closeCompletion = document.getElementById('close-completion');

      const playAnother = document.getElementById('play-another');

      closeCompletion.addEventListener('click', () => {

        completionModal.classList.add('hidden');

      });

      playAnother.addEventListener('click', () => {

        completionModal.classList.add('hidden');

        initGame();

      });


      // Vowel keys – clicking or dragging a key into a cell

      const vowelKeys = document.querySelectorAll('.vowel-key');

      vowelKeys.forEach(key => {

        // Desktop click and drag events remain the same

        key.addEventListener('click', () => {

          if (currentSelectedInput) {

            currentSelectedInput.value = key.dataset.letter;

            checkCompletion();

          }

        });

        key.addEventListener('dragstart', (e) => {

          e.dataTransfer.setData('text/plain', key.dataset.letter);

        });

        // On mobile, add touch events to simulate dragging

        if (isMobile()) {

          key.addEventListener('touchstart', onVowelTouchStart, {passive: false});

          key.addEventListener('touchmove', onVowelTouchMove, {passive: false});

          key.addEventListener('touchend', onVowelTouchEnd, {passive: false});

        }

      });


      // Button event listeners:

      // "New grid" picks a new grid and resets the timer

      document.getElementById('new-grid-button').addEventListener('click', initGame);

      // "Reset" clears the current grid's inputs and resets the timer

      document.getElementById('reset-button').addEventListener('click', resetCurrentGrid);


      // Start the game for the first time

      initGame();

    });
